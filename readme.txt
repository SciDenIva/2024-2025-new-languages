Структура директорий
[<root>]
  - [<language_name_1>] // директория для конкретного языка программирования
    - [doc] // различная документация
      - references.md - список летературы, статей и других материалов (пополняемый)
    - [01] // первая презентация
      - [src] // исходный код к презентации
      - <01>-01.pdf // файл с первой частью презентации
      - <01>-02.pdf // файл со второй частью презентации (опционально)
  - [<language_name-2>] // директория для конкретного языка программирования
    - ...
